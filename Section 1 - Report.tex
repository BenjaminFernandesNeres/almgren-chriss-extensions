\section{Model Setup}
% -------------------------------------------------------------------

We want to liquidate a block of $Q > 0$ shares over a fixed time horizon $[0,T]$.
The key quantities are:

\begin{itemize}[leftmargin=1.5em]
  \item $q_t$: the number of shares still held (\emph{inventory}) at time $t$.
  \item $v_t = -\dot q_t$: the \emph{selling speed}, i.e.\ how many shares we sell per unit of time.
    A positive $v_t$ means we are selling.
  \item Boundary conditions: $q_0 = Q$ (we start with everything) and $q_T = 0$ (we must finish flat). These two conditions together imply
        \[
          \int_0^T v_t \dd t = Q.
        \]
\end{itemize}

\paragraph{Price dynamics (no permanent impact).}
The unaffected mid-price follows a Brownian motion with no drift:
\[
  \dd S_t = \sigma \dd W_t, \qquad S_t = S_0 + \sigma W_t,
\]
where $\sigma > 0$ is the volatility and $W_t$ is a standard Brownian motion.

\paragraph{Temporary market impact.}
When we sell at speed $v_t$, we push the price against ourselves momentarily.
The actual execution price is
\[
  \tilde S_t = S_t - \eta v_t,
\]
where $\eta > 0$ is the temporary impact parameter.
Selling $v_t \dd t$ shares at this price generates cash:
\[
  \dd X_t = \tilde S_t \, v_t \dd t = (S_t - \eta v_t)\,v_t \dd t.
\]

\paragraph{Total cash received.}
Integrating over the whole horizon:
\[
  X_T = \int_0^T (S_t - \eta v_t)\,v_t \dd t.
\]

\paragraph{Criterion.}
We use a \emph{mean--variance} objective: for risk-aversion parameter $\lambda > 0$,
we minimise
\[
  J = -\E[\Pi] + \lambda\,\Var(\Pi),
\]
where $\Pi$ is the \emph{payoff} of the strategy relative to a chosen benchmark.
We restrict to \emph{static} (deterministic) strategies: $q_t$ and $v_t$ are
deterministic functions of time, so they are not adapted to the Brownian motion.

% -------------------------------------------------------------------
\section{A Shared Tool: Integration by Parts on the Cash Flow}
\label{sec:ibp}
% -------------------------------------------------------------------

Before treating each benchmark separately, we derive a representation of $X_T$
that will be used throughout.

\paragraph{Expanding the cash flow.}
Substituting $S_t = S_0 + \sigma W_t$:
\begin{align}
  X_T
  &= \int_0^T S_t\,v_t \dd t - \eta\int_0^T v_t^2 \dd t \notag\\
  &= S_0\!\int_0^T v_t \dd t
     + \sigma\!\int_0^T W_t\,v_t \dd t
     - \eta\!\int_0^T v_t^2 \dd t.
\end{align}
Using the boundary condition $\int_0^T v_t \dd t = Q$, the first term simplifies to $S_0 Q$:
\begin{equation}
  X_T = S_0 Q + \sigma\!\int_0^T W_t\,v_t \dd t - \eta\!\int_0^T v_t^2 \dd t.
  \label{eq:XT-raw}
\end{equation}

\paragraph{Stochastic integration by parts.}
The stochastic integral $\int_0^T W_t \,v_t \dd t$ is an ordinary (Lebesgue) integral
because $v_t$ is deterministic, but $W_t$ is random.
We convert it into a stochastic integral (with respect to $W$)
using integration by parts. Recall that $v_t = -\dot q_t$, so $v_t \dd t = -\dd q_t$.
We integrate by parts with
\[
  u = W_t, \quad \dd V = v_t \dd t = -\dd q_t:
\]
\[
  \int_0^T W_t\,v_t \dd t
  = -\int_0^T W_t \dd q_t
  = -\bigl[W_t\,q_t\bigr]_0^T + \int_0^T q_t \dd W_t.
\]
Evaluating the boundary term: $W_0 = 0$ and $q_T = 0$, so
\[
  \bigl[W_t\,q_t\bigr]_0^T = W_T\cdot 0 - 0\cdot Q = 0.
\]
Therefore:
\begin{equation}
  \boxed{\int_0^T W_t\,v_t \dd t = \int_0^T q_t \dd W_t.}
  \label{eq:ibp}
\end{equation}

\begin{keybox}[Why this matters]
  The left-hand side of \eqref{eq:ibp} mixes the deterministic function $v_t$
  with the Brownian motion $W_t$ in a way that is hard to work with directly.
  The right-hand side is a \emph{stochastic integral} driven by $\dd W_t$
  with the \textbf{deterministic} integrand $q_t$.
  For a deterministic integrand, It\^{o}'s isometry gives directly:
  \[
    \Var\!\left(\int_0^T q_t \dd W_t\right) = \int_0^T q_t^2 \dd t.
  \]
  This is the key that unlocks the variational formulation.
\end{keybox}

\paragraph{Canonical form of the cash flow.}
Substituting \eqref{eq:ibp} into \eqref{eq:XT-raw}:
\begin{equation}
  X_T = S_0 Q - \eta\!\int_0^T v_t^2 \dd t + \sigma\!\int_0^T q_t \dd W_t.
  \label{eq:XT-canonical}
\end{equation}
This form holds \emph{for any admissible strategy} and will be our starting point
in each of the three benchmark cases below.

% -------------------------------------------------------------------
\section{Implementation Shortfall (IS)}
\label{sec:IS}
% -------------------------------------------------------------------

\subsection{Benchmark and payoff}

The \emph{implementation shortfall} benchmark is the initial price $S_0$:
the ideal would be to sell all $Q$ shares instantly at $S_0$, receiving $QS_0$.
The shortfall is the gap between this ideal and what we actually collect:
\[
  \Pi_{\mathrm{IS}} = X_T - QS_0.
\]
Minimising the shortfall (in mean--variance sense) is equivalent to maximising the
cash $X_T$, so we use the objective
\[
  J_{\mathrm{IS}} = -\E[\Pi_{\mathrm{IS}}] + \lambda\,\Var(\Pi_{\mathrm{IS}}).
\]
Since $QS_0$ is a constant, it vanishes from both the mean and the variance:
$\E[\Pi_{\mathrm{IS}}] = \E[X_T] - QS_0$ and $\Var(\Pi_{\mathrm{IS}}) = \Var(X_T)$.

\subsection{Mean of \texorpdfstring{$X_T$}{XT}}

From the canonical form \eqref{eq:XT-canonical}, the only random term is
$\sigma\int_0^T q_t \dd W_t$. Since $q_t$ is deterministic and
$\E[\dd W_t] = 0$, the stochastic integral has zero expectation. Therefore:
\[
  \E[X_T] = S_0 Q - \eta\!\int_0^T v_t^2 \dd t.
\]
The term $-S_0 Q$ is a constant and can be dropped from the optimisation, leaving:
\[
  -\E[\Pi_{\mathrm{IS}}] = \eta\!\int_0^T v_t^2 \dd t = \eta\!\int_0^T \dot q_t^2 \dd t.
\]

\subsection{Variance of \texorpdfstring{$X_T$}{XT}}

The deterministic terms in \eqref{eq:XT-canonical} do not contribute to variance.
Applying It\^{o}'s isometry to the stochastic integral:
\[
  \Var(X_T)
  = \sigma^2\,\Var\!\left(\int_0^T q_t \dd W_t\right)
  = \sigma^2\!\int_0^T q_t^2 \dd t.
\]

\subsection{Variational formulation}

Combining mean and variance:
\begin{equation}
  J_{\mathrm{IS}}[q]
  = \eta\!\int_0^T \dot q_t^2 \dd t + \lambda\sigma^2\!\int_0^T q_t^2 \dd t,
  \label{eq:JIS}
\end{equation}
subject to $q_0 = Q$, $q_T = 0$.

\begin{notebox}[Reading the objective]
  The first term $\eta\int\dot q_t^2$ is the \emph{impact cost}: selling faster
  (larger $|\dot q_t|$) costs more. The second term $\lambda\sigma^2\int q_t^2$
  is the \emph{market risk}: holding more inventory (larger $|q_t|$) exposes us to
  more price uncertainty. The optimal schedule balances these two forces.
\end{notebox}

\subsection{Euler--Lagrange equation}

The Lagrangian of \eqref{eq:JIS} is $\mathcal{L}(q, \dot q) = \eta\dot q^2 + \lambda\sigma^2 q^2$.
The Euler--Lagrange equation
$\frac{d}{dt}\!\left(\frac{\partial\mathcal{L}}{\partial\dot q}\right)
 - \frac{\partial\mathcal{L}}{\partial q} = 0$
gives:
\[
  \frac{d}{dt}(2\eta\dot q) - 2\lambda\sigma^2 q = 0
  \implies
  \ddot q - \kap^2 q = 0,
  \qquad \kap^2 \coloneqq \frac{\lambda\sigma^2}{\eta}.
\]

\subsection{Solution}

The general solution of $\ddot q - \kap^2 q = 0$ is:
\[
  q(t) = A\cosh(\kap t) + B\sinh(\kap t).
\]
Applying the boundary conditions:

\textbf{At} $t = 0$: $q(0) = A = Q$, so $A = Q$.

\textbf{At} $t = T$: $Q\cosh(\kap T) + B\sinh(\kap T) = 0$, so
$B = -Q\dfrac{\cosh(\kap T)}{\sinh(\kap T)} = -Q\coth(\kap T)$.

Therefore $q(t) = Q\cosh(\kap t) - Q\coth(\kap T)\sinh(\kap t)$.
Using the hyperbolic identity
$\sinh(\kap(T-t)) = \sinh(\kap T)\cosh(\kap t) - \cosh(\kap T)\sinh(\kap t)$,
we can rewrite this more compactly as:

\begin{resultbox}[IS optimal schedule]
\[
  q_{\mathrm{IS}}(t) = Q\,\frac{\sinh\!\bigl(\kap(T-t)\bigr)}{\sinh(\kap T)},
  \qquad
  v_{\mathrm{IS}}(t) = Q\,\frac{\kap\cosh\!\bigl(\kap(T-t)\bigr)}{\sinh(\kap T)},
  \qquad \kap^2 = \frac{\lambda\sigma^2}{\eta}.
\]
For large $\kap$ (high risk aversion or high volatility), the schedule front-loads
selling: $q_t$ drops quickly then flattens. For small $\kap$, it is close to linear.
\end{resultbox}

\section{Target Close (TC)}
\label{sec:TC}
% -------------------------------------------------------------------

\subsection{Benchmark and payoff}

The \emph{target close} benchmark is the terminal price $S_T$: we measure our
performance relative to the price at the end of the trading window. The payoff is:
\[
  \Pi_{\mathrm{TC}} = X_T - QS_T.
\]

\subsection{Simplifying the payoff}

Starting from the canonical form \eqref{eq:XT-canonical} and writing
$S_T = S_0 + \sigma W_T$:
\begin{align*}
  \Pi_{\mathrm{TC}}
  &= S_0 Q - \eta\!\int_0^T v_t^2 \dd t + \sigma\!\int_0^T q_t \dd W_t
     - Q(S_0 + \sigma W_T).
\end{align*}
The $S_0 Q$ terms cancel. We express $W_T = \int_0^T \dd W_t$, so $QW_T = Q\int_0^T \dd W_t$:
\begin{align}
  \Pi_{\mathrm{TC}}
  &= - \eta\!\int_0^T v_t^2 \dd t + \sigma\!\int_0^T q_t \dd W_t
     - Q\sigma\!\int_0^T \dd W_t \notag\\
  &= - \eta\!\int_0^T v_t^2 \dd t + \sigma\!\int_0^T (q_t - Q)\,\dd W_t.
  \label{eq:PiTC}
\end{align}

\begin{keybox}[Structure of $\Pi_{\mathrm{TC}}$]
  Comparing \eqref{eq:PiTC} with the IS expression,
  the only difference is that the stochastic integrand changes from $q_t$ to $q_t - Q$.
  This reflects the change of benchmark: for TC we care about
  how much $q_t$ deviates from $Q$ (i.e.\ how much we have \emph{already sold}),
  whereas for IS we care about $q_t$ itself (how much we still \emph{hold}).
\end{keybox}

\subsection{Mean and variance}

\textbf{Mean.} The stochastic integral has zero expectation:
\[
  -\E[\Pi_{\mathrm{TC}}] = \eta\!\int_0^T v_t^2 \dd t = \eta\!\int_0^T \dot q_t^2 \dd t.
\]

\textbf{Variance.} By It\^{o}'s isometry:
\[
  \Var(\Pi_{\mathrm{TC}})
  = \sigma^2\!\int_0^T (q_t - Q)^2 \dd t.
\]

\subsection{Variational formulation}

\begin{equation}
  J_{\mathrm{TC}}[q]
  = \eta\!\int_0^T \dot q_t^2 \dd t + \lambda\sigma^2\!\int_0^T (q_t - Q)^2 \dd t,
  \label{eq:JTC}
\end{equation}
subject to $q_0 = Q$, $q_T = 0$.

\begin{notebox}[Reading the objective]
  Compared to IS, the risk term now penalises the deviation of $q_t$ from $Q$
  rather than from $0$. This makes sense: for a TC order, risk arises from
  having \emph{sold too early}, because we are short relative to the benchmark $S_T$.
  The optimal strategy therefore \emph{delays} selling compared to IS.
\end{notebox}

\subsection{Euler--Lagrange equation}

The Lagrangian is $\mathcal{L}_{\mathrm{TC}}(q, \dot q) = \eta\dot q^2 + \lambda\sigma^2(q - Q)^2$.
Computing the partial derivatives:
\[
  \frac{\partial\mathcal{L}_{\mathrm{TC}}}{\partial\dot q} = 2\eta\dot q,
  \qquad
  \frac{\partial\mathcal{L}_{\mathrm{TC}}}{\partial q} = 2\lambda\sigma^2(q - Q).
\]
The Euler--Lagrange equation gives:
\[
  \ddot q - \kap^2(q - Q) = 0, \qquad \kap^2 = \frac{\lambda\sigma^2}{\eta}.
\]

\subsection{Change of variable}

Define $y(t) = q(t) - Q$, which measures how far we are from a full position.
Then $\ddot y = \ddot q$, and the ODE becomes:
\[
  \ddot y - \kap^2 y = 0.
\]
The boundary conditions translate to:
\[
  y(0) = q(0) - Q = Q - Q = 0, \qquad y(T) = q(T) - Q = 0 - Q = -Q.
\]

\subsection{Solution}

General solution: $y(t) = A\cosh(\kap t) + B\sinh(\kap t)$.

\textbf{At} $t = 0$: $y(0) = A = 0$.

\textbf{At} $t = T$: $B\sinh(\kap T) = -Q$, so $B = -Q/\sinh(\kap T)$.

Therefore $y(t) = -Q\sinh(\kap t)/\sinh(\kap T)$,
and recalling $q(t) = Q + y(t)$:

\begin{resultbox}[TC optimal schedule]
\[
  q_{\mathrm{TC}}(t) = Q\!\left(1 - \frac{\sinh(\kap t)}{\sinh(\kap T)}\right),
  \qquad
  v_{\mathrm{TC}}(t) = Q\,\frac{\kap\cosh(\kap t)}{\sinh(\kap T)},
  \qquad \kap^2 = \frac{\lambda\sigma^2}{\eta}.
\]
The schedule starts slowly (holding most of the position)
and accelerates toward the end --- the mirror image of IS.
\end{resultbox}

\section{TWAP Benchmark}
\label{sec:TWAP}
% -------------------------------------------------------------------

\subsection{Benchmark and payoff}

The \emph{time-weighted average price} benchmark is the time-average of the mid-price:
\[
  B_{\mathrm{TWAP}} = \frac{1}{T}\int_0^T S_t \dd t.
\]
The payoff is:
\[
  \Pi_{\mathrm{TWAP}} = X_T - Q \cdot B_{\mathrm{TWAP}}
  = X_T - \frac{Q}{T}\int_0^T S_t \dd t.
\]

\subsection{Handling the benchmark term}

We need to simplify $\frac{Q}{T}\int_0^T S_t \dd t$.
Substituting $S_t = S_0 + \sigma W_t$:
\[
  \frac{Q}{T}\int_0^T S_t \dd t
  = \frac{Q}{T}\!\int_0^T S_0 \dd t + \frac{Q\sigma}{T}\!\int_0^T W_t \dd t
  = QS_0 + \frac{Q\sigma}{T}\!\int_0^T W_t \dd t.
\]

We now simplify the remaining integral $\int_0^T W_t \dd t$
using integration by parts. Set $u = W_t$ and $\dd V = \dd t$,
so that $\dd u = \dd W_t$ and $V = t$:
\begin{align}
  \int_0^T W_t \dd t
  &= \Bigl[t\, W_t\Bigr]_0^T - \int_0^T t\, \dd W_t
   = T W_T - \int_0^T t\, \dd W_t.
   \label{eq:ibp-twap}
\end{align}
Writing $W_T = \int_0^T \dd W_t$, the first term becomes
$T W_T = T\int_0^T \dd W_t = \int_0^T T\,\dd W_t$,
so the two terms combine into a single stochastic integral:
\[
  \int_0^T W_t \dd t
  = \int_0^T (T - t)\,\dd W_t.
\]

Therefore:
\[
  \frac{Q}{T}\int_0^T S_t \dd t
  = QS_0 + \frac{Q\sigma}{T}\int_0^T (T-t)\,\dd W_t.
\]

\begin{keybox}[The natural benchmark trajectory]
  Define the \emph{deterministic linear schedule}
  \[
    f(t) \coloneqq Q\,\frac{T-t}{T},
  \]
  which starts at $Q$ when $t=0$ and reaches $0$ at $t=T$:
  it is the inventory path of a trader selling at a perfectly uniform rate $Q/T$.
  Equation \eqref{eq:PiTWAP} shows that the stochastic risk of a TWAP order
  is entirely governed by the \emph{deviation} $q_t - f(t)$.
  This requires no time-average of $q$ — the benchmark enters directly as a path.
\end{keybox}


\subsection{Mean and variance}

\textbf{Mean.} The stochastic integral has zero expectation:
\[
  -\E[\Pi_{\mathrm{TWAP}}] = \eta\!\int_0^T v_t^2 \dd t = \eta\!\int_0^T \dot q_t^2 \dd t.
\]

\textbf{Variance.} By It\^{o}'s isometry applied to \eqref{eq:PiTWAP}:
\[
  \Var(\Pi_{\mathrm{TWAP}})
  = \sigma^2\!\int_0^T \!\left(q_t - Q\,\frac{T-t}{T}\right)^{\!2} \dd t.
\]

\subsection{Variational formulation}

\begin{equation}
  J_{\mathrm{TWAP}}[q]
  = \eta\!\int_0^T \dot q_t^2 \dd t
    + \lambda\sigma^2\!\int_0^T \!\left(q_t - Q\,\frac{T-t}{T}\right)^{\!2} \dd t,
  \label{eq:JTWAP}
\end{equation}
subject to $q_0 = Q$, $q_T = 0$.

\begin{notebox}[Comparison with IS and TC]
The three objectives have identical structure:
\[
  J[q] = \eta\!\int_0^T\dot q_t^2\,\dd t + \lambda\sigma^2\!\int_0^T\bigl(q_t - f(t)\bigr)^2\dd t,
\]
where the ``target path'' $f$ changes by benchmark:
\begin{center}
  \renewcommand{\arraystretch}{1.4}
  \begin{tabular}{@{}lll@{}}
    \toprule
    Benchmark & Target path $f(t)$ & Intuition \\
    \midrule
    IS   & $0$              & Hold nothing; sell everything up front. \\
    TC   & $Q$              & Hold everything; sell everything at the end. \\
    TWAP & $Q(T-t)/T$  & Hold and sell at a constant linear rate. \\
    \bottomrule
  \end{tabular}
\end{center}
\end{notebox}

\subsection{Euler--Lagrange equation}

The Lagrangian is $\mathcal{L}_{\mathrm{TWAP}}(q,\dot q, t) = \eta\dot q^2 + \lambda\sigma^2(q - f(t))^2$,
where $f(t) = Q(T-t)/T$ is a \emph{known, deterministic} function of time.
Computing the partial derivatives:
\[
  \frac{\partial\mathcal{L}_{\mathrm{TWAP}}}{\partial\dot q} = 2\eta\dot q,
  \qquad
  \frac{\partial\mathcal{L}_{\mathrm{TWAP}}}{\partial q} = 2\lambda\sigma^2(q - f(t)).
\]
The Euler--Lagrange equation gives:
\[
  \ddot q - \kap^2\!\left(q - Q\,\frac{T-t}{T}\right) = 0, \qquad \kap^2 = \frac{\lambda\sigma^2}{\eta}.
\]

\subsection{Change of variable}

Define the deviation from the linear schedule:
\[
  y(t) \coloneqq q(t) - f(t) = q(t) - Q\,\frac{T-t}{T}.
\]
Since $f$ is linear, its second derivative vanishes: $\ddot f(t) = 0$.
Therefore $\ddot y = \ddot q$, and the ODE reduces to:
\[
  \ddot y - \kap^2 y = 0.
\]
Crucially, this is the same ODE as before — but the boundary conditions are now:
\[
  y(0) = q(0) - f(0) = Q - Q = 0, \qquad y(T) = q(T) - f(T) = 0 - 0 = 0.
\]
\emph{Both} boundary values are zero!

\subsection{Solution}

General solution: $y(t) = A\cosh(\kap t) + B\sinh(\kap t)$.

\textbf{At} $t = 0$: $y(0) = A = 0$.

\textbf{At} $t = T$: $B\sinh(\kap T) = 0$.

Since $\kap > 0$, we have $\sinh(\kap T) > 0$, which forces $B = 0$.

Therefore $y(t) = 0$ for all $t \in [0, T]$, meaning $q(t) = f(t)$ exactly:

\begin{resultbox}[TWAP optimal schedule --- exact result]
\[
  q_{\mathrm{TWAP}}(t) = Q\!\left(1 - \frac{t}{T}\right),
  \qquad
  v_{\mathrm{TWAP}}(t) = \frac{Q}{T}.
\]
The optimal strategy is to sell at a \textbf{perfectly constant rate} $Q/T$
for all $t$, for \textbf{any} value of $\lambda > 0$ and any $\kap$.
This is an exact result, not an approximation.
\end{resultbox}

\paragraph{Why the result is exact.}
The function $f(t) = Q(T-t)/T$ already satisfies both boundary conditions
$f(0) = Q$ and $f(T) = 0$, so it is a \emph{feasible} trajectory.
Since the objective \eqref{eq:JTWAP} is a sum of squares of $\dot q_t$ and
$(q_t - f(t))$, it is minimised at zero when $q_t = f(t)$: the deviation
$y(t) = 0$ simultaneously zeroes out the risk term.
There is no trade-off to resolve — selling linearly at the TWAP rate
is both the cheapest (in impact) and the safest (in risk) thing to do
within the TWAP benchmark.

\section{Summary}
\label{sec:summary}
% -------------------------------------------------------------------

\begin{resultbox}[Three optimal schedules]
All three strategies share the same Euler--Lagrange structure,
$\ddot y - \kap^2 y = 0$ on the deviation $y(t) = q(t) - f(t)$,
with $\kap^2 = \lambda\sigma^2/\eta$.

\medskip
\begin{center}
  \renewcommand{\arraystretch}{2.0}
  \begin{tabular}{@{}llll@{}}
    \toprule
    \textbf{Benchmark} & \textbf{Optimal inventory} $q(t)$ & \textbf{BCs on $y$} & \textbf{$B$} \\
    \midrule
    IS & $\displaystyle Q\frac{\sinh\kap(T-t)}{\sinh\kap T}$
       & $y(0)=Q,\; y(T)=0$
       & $-Q\coth\kap T$ \\[8pt]
    TC & $\displaystyle Q\!\left(1 - \frac{\sinh\kap t}{\sinh\kap T}\right)$
       & $y(0)=0,\; y(T)=-Q$
       & $\displaystyle -\frac{Q}{\sinh\kap T}$ \\[8pt]
    TWAP & $\displaystyle Q\!\left(1 - \frac{t}{T}\right)$
         & $y(0)=0,\; y(T)=0$
         & $0$ \\
    \bottomrule
  \end{tabular}
\end{center}

\medskip
The stochastic risk of each strategy, identified from the payoff via
integration by parts and It\^{o}'s isometry, is:
\[
  \Var(\Pi)
  = \sigma^2\!\int_0^T\!\bigl(q_t - f(t)\bigr)^2\dd t,
  \quad\text{where }
  f(t) = \begin{cases} 0 & \text{(IS)} \\ Q & \text{(TC)} \\ Q(T-t)/T & \text{(TWAP)} \end{cases}
\]
\end{resultbox}

\section{Efficient Frontiers for IS and TC}
% -------------------------------------------------------------------

\subsection{Implementation Shortfall}

\paragraph{Cost.}
Recall the optimal IS trading speed:
\[
  v_{\mathrm{IS}}(t) = Q\,\frac{\kap\cosh\!\bigl(\kap(T-t)\bigr)}{\sinh(\kap T)}.
\]
We compute:
\begin{align*}
  \mathrm{Cost}_{\mathrm{IS}}
  &= \eta\int_0^T v_{\mathrm{IS}}(t)^2\,\dd t
   = \eta Q^2\,\frac{\kap^2}{\sinh^2(\kap T)}\int_0^T \cosh^2\!\bigl(\kap(T-t)\bigr)\,\dd t.
\end{align*}
Substituting $u = T - t$:
\[
  \int_0^T \cosh^2\!\bigl(\kap(T-t)\bigr)\,\dd t
  = \int_0^T \cosh^2(\kap u)\,\dd u.
\]
Using the identity $\cosh^2(\kap u) = \dfrac{1 + \cosh(2\kap u)}{2}$:
\[
  \int_0^T \cosh^2(\kap u)\,\dd u
  = \frac{T}{2} + \frac{1}{2}\int_0^T \cosh(2\kap u)\,\dd u
  = \frac{T}{2} + \frac{\sinh(2\kap T)}{4\kap}.
\]
Using the identity $\sinh(2\kap T) = 2\sinh(\kap T)\cosh(\kap T)$:
\[
  \int_0^T \cosh^2(\kap u)\,\dd u
  = \frac{T}{2} + \frac{\sinh(\kap T)\cosh(\kap T)}{2\kap}.
\]
Therefore:
\begin{align*}
  \mathrm{Cost}_{\mathrm{IS}}
  &= \frac{\eta Q^2 \kap^2}{\sinh^2(\kap T)}
     \left(\frac{T}{2} + \frac{\sinh(\kap T)\cosh(\kap T)}{2\kap}\right) \\
  &= \frac{\eta Q^2 \kap^2}{2\sinh^2(\kap T)}
     \left(T + \frac{\sinh(\kap T)\cosh(\kap T)}{\kap}\right) \\
  &= \frac{\eta Q^2 \kap}{2}
     \left(\frac{\kap T}{\sinh^2(\kap T)} + \frac{\cosh(\kap T)}{\sinh(\kap T)}\right) \\
  &= \frac{\eta Q^2 \kap}{2}
     \left(\frac{\kap T}{\sinh^2(\kap T)} + \coth(\kap T)\right).
\end{align*}

\begin{resultbox}[IS cost --- closed form]
\[
  \mathrm{Cost}_{\mathrm{IS}}
  = \frac{\eta Q^2 \kap}{2}
    \left(\coth(\kap T) + \frac{\kap T}{\sinh^2(\kap T)}\right).
\]
\end{resultbox}

\paragraph{Risk.}
Recall the optimal IS inventory:
\[
  q_{\mathrm{IS}}(t) = Q\,\frac{\sinh\!\bigl(\kap(T-t)\bigr)}{\sinh(\kap T)}.
\]
We compute:
\begin{align*}
  \mathrm{Risk}_{\mathrm{IS}}
  &= \sigma^2\int_0^T q_{\mathrm{IS}}(t)^2\,\dd t
   = \sigma^2 Q^2\,\frac{1}{\sinh^2(\kap T)}\int_0^T \sinh^2\!\bigl(\kap(T-t)\bigr)\,\dd t.
\end{align*}
Substituting $u = T - t$:
\[
  \int_0^T \sinh^2\!\bigl(\kap(T-t)\bigr)\,\dd t
  = \int_0^T \sinh^2(\kap u)\,\dd u.
\]
Using the identity $\sinh^2(\kap u) = \dfrac{\cosh(2\kap u) - 1}{2}$:
\[
  \int_0^T \sinh^2(\kap u)\,\dd u
  = \frac{1}{2}\int_0^T \cosh(2\kap u)\,\dd u - \frac{T}{2}
  = \frac{\sinh(2\kap T)}{4\kap} - \frac{T}{2}.
\]
Using $\sinh(2\kap T) = 2\sinh(\kap T)\cosh(\kap T)$:
\[
  \int_0^T \sinh^2(\kap u)\,\dd u
  = \frac{\sinh(\kap T)\cosh(\kap T)}{2\kap} - \frac{T}{2}.
\]
Therefore:
\begin{align*}
  \mathrm{Risk}_{\mathrm{IS}}
  &= \frac{\sigma^2 Q^2}{\sinh^2(\kap T)}
     \left(\frac{\sinh(\kap T)\cosh(\kap T)}{2\kap} - \frac{T}{2}\right) \\
  &= \frac{\sigma^2 Q^2}{2\kap}
     \left(\frac{\cosh(\kap T)}{\sinh(\kap T)} - \frac{\kap T}{\sinh^2(\kap T)}\right) \\
  &= \frac{\sigma^2 Q^2}{2\kap}
     \left(\coth(\kap T) - \frac{\kap T}{\sinh^2(\kap T)}\right).
\end{align*}

\begin{resultbox}[IS risk --- closed form]
\[
  \mathrm{Risk}_{\mathrm{IS}}
  = \frac{\sigma^2 Q^2}{2\kap}
    \left(\coth(\kap T) - \frac{\kap T}{\sinh^2(\kap T)}\right).
\]
\end{resultbox}

% -------------------------------------------------------------------
\subsection{Target Close}
% -------------------------------------------------------------------

\paragraph{Cost.}
Recall the optimal TC trading speed:
\[
  v_{\mathrm{TC}}(t) = Q\,\frac{\kap\cosh(\kap t)}{\sinh(\kap T)}.
\]
We compute:
\begin{align*}
  \mathrm{Cost}_{\mathrm{TC}}
  &= \eta\int_0^T v_{\mathrm{TC}}(t)^2\,\dd t
   = \eta Q^2\,\frac{\kap^2}{\sinh^2(\kap T)}\int_0^T \cosh^2(\kap t)\,\dd t.
\end{align*}
The integral $\int_0^T \cosh^2(\kap t)\,\dd t$ is identical to the one computed in the IS
case (with $u = t$ instead of $u = T-t$):
\[
  \int_0^T \cosh^2(\kap t)\,\dd t
  = \frac{T}{2} + \frac{\sinh(\kap T)\cosh(\kap T)}{2\kap}.
\]
Therefore, the computation is word-for-word identical to the IS cost, giving:

\begin{resultbox}[TC cost --- closed form]
\[
  \mathrm{Cost}_{\mathrm{TC}}
  = \frac{\eta Q^2 \kap}{2}
    \left(\coth(\kap T) + \frac{\kap T}{\sinh^2(\kap T)}\right)
  = \mathrm{Cost}_{\mathrm{IS}}.
\]
\end{resultbox}

\paragraph{Risk.}
For TC the risk integrand is $(q_t - Q)^2$. Recall:
\[
  q_{\mathrm{TC}}(t) - Q = -Q\,\frac{\sinh(\kap t)}{\sinh(\kap T)}.
\]
We compute:
\begin{align*}
  \mathrm{Risk}_{\mathrm{TC}}
  &= \sigma^2\int_0^T \bigl(q_{\mathrm{TC}}(t) - Q\bigr)^2\,\dd t
   = \sigma^2 Q^2\,\frac{1}{\sinh^2(\kap T)}\int_0^T \sinh^2(\kap t)\,\dd t.
\end{align*}
Again, the integral $\int_0^T \sinh^2(\kap t)\,\dd t$ is identical to the one computed
in the IS case:
\[
  \int_0^T \sinh^2(\kap t)\,\dd t
  = \frac{\sinh(\kap T)\cosh(\kap T)}{2\kap} - \frac{T}{2}.
\]
Therefore:

\begin{resultbox}[TC risk --- closed form]
\[
  \mathrm{Risk}_{\mathrm{TC}}
  = \frac{\sigma^2 Q^2}{2\kap}
    \left(\coth(\kap T) - \frac{\kap T}{\sinh^2(\kap T)}\right)
  = \mathrm{Risk}_{\mathrm{IS}}.
\]
\end{resultbox}

% -------------------------------------------------------------------
\subsection{Why the Frontiers Coincide}
% -------------------------------------------------------------------

The equality of cost and risk for IS and TC is not a coincidence.
The two schedules are related by a time-reversal:
\[
  q_{\mathrm{TC}}(t) = Q - q_{\mathrm{IS}}(T - t).
\]
Taking the derivative: $v_{\mathrm{TC}}(t) = v_{\mathrm{IS}}(T-t)$,
so the cost integrands are related by $t \mapsto T - t$ and integrate to the same value.
For the risk, note that
\[
  q_{\mathrm{TC}}(t) - Q = -q_{\mathrm{IS}}(T-t),
\]
so the risk integrands are also related by $t \mapsto T - t$ and again integrate identically.

\begin{resultbox}[Identical efficient frontiers]
The IS and TC efficient frontiers coincide exactly:
\[
  \mathrm{Cost}^\star(\kap) = \frac{\eta Q^2 \kap}{2}
    \left(\coth(\kap T) + \frac{\kap T}{\sinh^2(\kap T)}\right)
    \]
  
\[
  \mathrm{Risk}^\star(\kap) = \frac{\sigma^2 Q^2}{2\kap}
    \left(\coth(\kap T) - \frac{\kap T}{\sinh^2(\kap T)}\right),
\]
for all $\kap = \sqrt{\lambda\sigma^2/\eta} > 0$.
\end{resultbox}

\section{Expected Utility Formulation for TC and TWAP}
\label{sec:bonus-expected-utility}
% -------------------------------------------------------------------

This section answers the bonus question in the expected-utility framework for
the \emph{Target Close} (TC) and \emph{TWAP} benchmarks only.
We use the same model and notation as in the previous sections, and we avoid
repeating the cash-flow and payoff derivations already established above.

\begin{keybox}[State-variable caution]
  The two HJB problems do not have the same primitive state variables:
  \begin{itemize}[leftmargin=1.5em]
    \item for TC, the state is $(t,x,s,q)$ because the benchmark depends on the
          terminal price $S_T$;
    \item for TWAP, the state is $(t,x,s,a,q)$, where
          \[
            A_t \coloneqq \frac{1}{T}\int_0^t S_u \dd u,
            \qquad
            \dd A_t = \frac{S_t}{T}\dd t,
          \]
          because the benchmark depends on the whole price path.
  \end{itemize}
\end{keybox}

\subsection{Expected utility criterion}

We replace the mean--variance objective by CARA utility:
\[
  U(x) = -e^{-\gamma x},
  \qquad \gamma > 0.
\]
For a benchmark-dependent payoff $\Pi_B$, the optimization problem is
\[
  \max_q \E[-e^{-\gamma \Pi_B}],
  \qquad
  B \in \{\mathrm{TC}, \mathrm{TWAP}\}.
\]

% -------------------------------------------------------------------
\subsection{Target Close (TC)}
% -------------------------------------------------------------------

\paragraph{Value function and HJB.}
For the TC benchmark, the payoff is
\[
  \Pi_{\mathrm{TC}} = X_T - QS_T.
\]
The value function is therefore
\begin{equation}
  V^{\mathrm{TC}}(t,x,s,q)
  =
  \sup_{(v_u)_{u \in [t,T]}}
  \E_{t,x,s,q}\!\left[-e^{-\gamma(X_T - QS_T)}\right].
  \label{eq:V-TC-bonus}
\end{equation}
Using the dynamics already introduced in the report,
\[
  \dd X_t = v_t(S_t-\eta v_t)\dd t,
  \qquad
  \dd S_t = \sigma \dd W_t,
  \qquad
  \dd q_t = -v_t \dd t,
\]
the HJB equation is
\begin{equation}
  0
  =
  V_t^{\mathrm{TC}}
  +
  \frac{1}{2}\sigma^2 V_{ss}^{\mathrm{TC}}
  +
  \sup_v
  \left\{
    v(s-\eta v)V_x^{\mathrm{TC}} - vV_q^{\mathrm{TC}}
  \right\}.
  \label{eq:HJB-TC-bonus}
\end{equation}

\paragraph{Exponential ansatz.}
Since $q_T = 0$, the terminal payoff can be rewritten as
\[
  X_T - QS_T = X_T + (q_T - Q)S_T.
\]
This suggests the ansatz
\begin{equation}
  V^{\mathrm{TC}}(t,x,s,q)
  =
  -\exp\!\left(
    -\gamma\bigl(x+(q-Q)s-\phi^{\mathrm{TC}}(t,q)\bigr)
  \right).
  \label{eq:ansatz-TC-bonus}
\end{equation}
The derivatives are
\[
  V_t^{\mathrm{TC}} = \gamma \phi_t^{\mathrm{TC}} V,
  \qquad
  V_x^{\mathrm{TC}} = -\gamma V,
\]
\[
  V_q^{\mathrm{TC}} = -\gamma(s-\phi_q^{\mathrm{TC}})V,
  \qquad
  V_s^{\mathrm{TC}} = -\gamma(q-Q)V,
\]
\[
  V_{ss}^{\mathrm{TC}} = \gamma^2(q-Q)^2V.
\]

\paragraph{Reduced HJB.}
Substituting these derivatives into \eqref{eq:HJB-TC-bonus} gives
\[
  \gamma \phi_t^{\mathrm{TC}} V
  +
  \frac{1}{2}\sigma^2\gamma^2(q-Q)^2V
  +
  \sup_v
  \left\{
    (vs-\eta v^2)(-\gamma V)
    -
    v\bigl(-\gamma(s-\phi_q^{\mathrm{TC}})V\bigr)
  \right\}
  = 0.
\]
Expanding the control term,
\[
  (vs-\eta v^2)(-\gamma V)
  -
  v\bigl(-\gamma(s-\phi_q^{\mathrm{TC}})V\bigr)
  =
  \gamma\eta v^2V - \gamma v\phi_q^{\mathrm{TC}}V.
\]
Since $-\gamma V > 0$, dividing through by $-\gamma V$ yields
\begin{equation}
  -\phi_t^{\mathrm{TC}}
  -
  \frac{\gamma\sigma^2}{2}(q-Q)^2
  +
  \sup_v
  \left\{
    v\phi_q^{\mathrm{TC}} - \eta v^2
  \right\}
  = 0.
  \label{eq:reduced-HJB-TC-bonus}
\end{equation}
The first-order condition is
\[
  \phi_q^{\mathrm{TC}} - 2\eta v = 0,
  \qquad
  v_{\mathrm{TC}}^*(t,q) = \frac{\phi_q^{\mathrm{TC}}(t,q)}{2\eta}.
\]
Hence the reduced HJB can be written as
\begin{equation}
  \phi_t^{\mathrm{TC}}
  +
  \inf_v
  \left\{
    \eta v^2 - v\phi_q^{\mathrm{TC}} + \frac{\gamma\sigma^2}{2}(q-Q)^2
  \right\}
  = 0.
  \label{eq:TC-running-cost-bonus}
\end{equation}
So the induced running cost is
\[
  \eta v_t^2 + \frac{\gamma\sigma^2}{2}(q_t-Q)^2.
\]
More precisely, the reduced HJB is now in the standard deterministic form
\[
  \phi_t + \inf_v\{L(q_t,v_t) + b(q_t,v_t)\phi_q\} = 0,
\]
with state dynamics
\[
  \dot q_t = b(q_t,v_t) = -v_t
\]
and running cost
\[
  L(q_t,v_t)
  =
  \eta v_t^2 + \frac{\gamma\sigma^2}{2}(q_t-Q)^2.
\]
Hence the associated deterministic control problem is to minimize
\[
  \int_0^T L(q_t,v_t)\dd t
\]
subject to $\dot q_t=-v_t$. Replacing $v_t$ by $-\dot q_t$ gives an objective
written entirely in terms of the inventory path $q$.

\paragraph{Equivalent deterministic objective.}
From the payoff representation derived earlier in the report,
\[
  \Pi_{\mathrm{TC}}
  =
  -\eta\!\int_0^T v_t^2 \dd t
  +
  \sigma\!\int_0^T (q_t-Q)\dd W_t.
\]
Because the strategy is deterministic, $\Pi_{\mathrm{TC}}$ is Gaussian with
\[
  \E[\Pi_{\mathrm{TC}}]
  =
  -\eta\!\int_0^T v_t^2 \dd t,
  \qquad
  \Var(\Pi_{\mathrm{TC}})
  =
  \sigma^2\!\int_0^T (q_t-Q)^2 \dd t.
\]
Since $\Pi_{\mathrm{TC}}$ is Gaussian and the utility is CARA,
\[
  \E[-e^{-\gamma \Pi_{\mathrm{TC}}}]
  =
  -\exp\!\left(
    -\gamma \E[\Pi_{\mathrm{TC}}]
    +
    \frac{\gamma^2}{2}\Var(\Pi_{\mathrm{TC}})
  \right).
\]
Because the exponential function is monotone, maximizing expected utility is
equivalent to maximizing
\[
  \E[\Pi_{\mathrm{TC}}] - \frac{\gamma}{2}\Var(\Pi_{\mathrm{TC}}),
\]
or equivalently minimizing
\[
  -\E[\Pi_{\mathrm{TC}}] + \frac{\gamma}{2}\Var(\Pi_{\mathrm{TC}}).
\]
Therefore maximizing expected utility is equivalent to minimizing
\begin{equation}
  J_{\mathrm{TC}}^{\mathrm{EU}}[q]
  =
  \eta\!\int_0^T \dot q_t^2 \dd t
  +
  \frac{\gamma\sigma^2}{2}\!\int_0^T (q_t-Q)^2 \dd t.
  \label{eq:J-TC-bonus}
\end{equation}

\paragraph{Optimal TC strategy.}
The Euler--Lagrange equation associated with \eqref{eq:J-TC-bonus} is
\[
  \frac{\dd}{\dd t}(2\eta \dot q_t) - \gamma\sigma^2(q_t-Q) = 0,
\]
that is,
\begin{equation}
  \ddot q_t = \kap^2(q_t-Q),
  \qquad
  \kap^2 = \frac{\gamma\sigma^2}{2\eta}.
  \label{eq:EL-TC-bonus}
\end{equation}
Let $y_t \coloneqq q_t-Q$. Then
\[
  \ddot y_t = \kap^2 y_t,
  \qquad
  y_0 = 0,
  \qquad
  y_T = -Q.
\]
The general solution is
\[
  y_t = A\sinh(\kap t) + B\cosh(\kap t).
\]
Since $y_0 = 0$, we get $B=0$. Since $y_T = -Q$, we get
\[
  A = -\frac{Q}{\sinh(\kap T)}.
\]
Therefore
\[
  y_t = -Q\frac{\sinh(\kap t)}{\sinh(\kap T)},
\]
and the optimal TC inventory is
\begin{equation}
  \boxed{
    q_{\mathrm{TC}}^*(t)
    =
    Q\left(1-\frac{\sinh(\kap t)}{\sinh(\kap T)}\right).
  }
  \label{eq:qstar-TC-bonus}
\end{equation}

% -------------------------------------------------------------------
\subsection{TWAP}
% -------------------------------------------------------------------

\paragraph{Value function and HJB.}
For TWAP, the benchmark is the running time average
\[
  A_T = \frac{1}{T}\int_0^T S_u \dd u,
\]
so the payoff is
\[
  \Pi_{\mathrm{TWAP}} = X_T - QA_T.
\]
The value function is
\begin{equation}
  V^{\mathrm{TWAP}}(t,x,s,a,q)
  =
  \sup_{(v_u)_{u \in [t,T]}}
  \E_{t,x,s,a,q}\!\left[-e^{-\gamma(X_T-QA_T)}\right].
  \label{eq:V-TWAP-bonus}
\end{equation}
The HJB equation now includes the extra state variable $a=A_t$:
\begin{equation}
  0
  =
  V_t^{\mathrm{TWAP}}
  +
  \frac{1}{2}\sigma^2 V_{ss}^{\mathrm{TWAP}}
  +
  \frac{s}{T}V_a^{\mathrm{TWAP}}
  +
  \sup_v
  \left\{
    v(s-\eta v)V_x^{\mathrm{TWAP}} - vV_q^{\mathrm{TWAP}}
  \right\}.
  \label{eq:HJB-TWAP-bonus}
\end{equation}

\paragraph{Exponential ansatz.}
To match the terminal payoff $X_T-QA_T$, we use
\begin{equation}
  V^{\mathrm{TWAP}}(t,x,s,a,q)
  =
  -\exp\!\left(
    -\gamma\left(
      x+\left(q-Q\frac{T-t}{T}\right)s-Qa-\phi^{\mathrm{TWAP}}(t,q)
    \right)
  \right).
  \label{eq:ansatz-TWAP-bonus}
\end{equation}
At time $T$, the coefficient of $s$ vanishes, so this indeed becomes
$x-Qa$.

From \eqref{eq:ansatz-TWAP-bonus}, the derivatives are
\[
  V_t^{\mathrm{TWAP}}
  =
  \gamma\left(\phi_t^{\mathrm{TWAP}}-\frac{Q}{T}s\right)V,
\]
\[
  V_x^{\mathrm{TWAP}} = -\gamma V,
  \qquad
  V_q^{\mathrm{TWAP}} = -\gamma(s-\phi_q^{\mathrm{TWAP}})V,
\]
\[
  V_s^{\mathrm{TWAP}}
  =
  -\gamma\left(q-Q\frac{T-t}{T}\right)V,
  \qquad
  V_{ss}^{\mathrm{TWAP}}
  =
  \gamma^2\left(q-Q\frac{T-t}{T}\right)^2V,
\]
\[
  V_a^{\mathrm{TWAP}} = \gamma QV.
\]

\paragraph{Reduced HJB.}
Substituting these derivatives into \eqref{eq:HJB-TWAP-bonus}, the terms
involving $Qs/T$ cancel. Dividing through by $-\gamma V > 0$ yields
\begin{equation}
  -\phi_t^{\mathrm{TWAP}}
  -
  \frac{\gamma\sigma^2}{2}
  \left(q-Q\frac{T-t}{T}\right)^2
  +
  \sup_v
  \left\{
    v\phi_q^{\mathrm{TWAP}} - \eta v^2
  \right\}
  = 0.
  \label{eq:reduced-HJB-TWAP-bonus}
\end{equation}
Equivalently,
\begin{equation}
  \phi_t^{\mathrm{TWAP}}
  +
  \inf_v
  \left\{
    \eta v^2
    -
    v\phi_q^{\mathrm{TWAP}}
    +
    \frac{\gamma\sigma^2}{2}
    \left(q-Q\frac{T-t}{T}\right)^2
  \right\}
  = 0.
  \label{eq:TWAP-running-cost-bonus}
\end{equation}
So the induced running cost is
\[
  \eta v_t^2
  +
  \frac{\gamma\sigma^2}{2}
  \left(q_t-Q\frac{T-t}{T}\right)^2.
\]
Again, this is a deterministic HJB of the form
\[
  \phi_t + \inf_v\{L(q_t,v_t) + b(q_t,v_t)\phi_q\} = 0,
\]
with
\[
  \dot q_t = b(q_t,v_t) = -v_t
\]
and
\[
  L(q_t,v_t)
  =
  \eta v_t^2
  +
  \frac{\gamma\sigma^2}{2}
  \left(q_t-Q\frac{T-t}{T}\right)^2.
\]
Therefore the associated deterministic control problem is to minimize
\[
  \int_0^T L(q_t,v_t)\dd t
\]
subject to $\dot q_t=-v_t$. Replacing $v_t$ by $-\dot q_t$ gives the
deterministic objective below.

\paragraph{Equivalent deterministic objective.}
From the payoff representation already obtained in the report,
\[
  \Pi_{\mathrm{TWAP}}
  =
  -\eta\!\int_0^T v_t^2 \dd t
  +
  \sigma\!\int_0^T
  \left(q_t-Q\frac{T-t}{T}\right)\dd W_t.
\]
Since the strategy is deterministic, $\Pi_{\mathrm{TWAP}}$ is Gaussian and
maximizing expected utility is equivalent to minimizing
\begin{equation}
  J_{\mathrm{TWAP}}^{\mathrm{EU}}[q]
  =
  \eta\!\int_0^T \dot q_t^2 \dd t
  +
  \frac{\gamma\sigma^2}{2}\!\int_0^T
  \left(q_t-Q\frac{T-t}{T}\right)^2 \dd t.
  \label{eq:J-TWAP-bonus}
\end{equation}

\paragraph{Optimal TWAP strategy.}
The Euler--Lagrange equation associated with \eqref{eq:J-TWAP-bonus} is
\begin{equation}
  \ddot q_t
  =
  \kap^2\left(q_t-Q\frac{T-t}{T}\right).
  \label{eq:EL-TWAP-bonus}
\end{equation}
Define the deviation from the TWAP benchmark path by
\[
  y_t \coloneqq q_t-Q\frac{T-t}{T}.
\]
Since $Q(T-t)/T$ is linear in $t$, its second derivative is zero, so
\[
  \ddot y_t = \kap^2 y_t.
\]
The boundary conditions become
\[
  y_0 = q_0-Q = 0,
  \qquad
  y_T = q_T-0 = 0.
\]
The only solution is therefore
\[
  y_t \equiv 0.
\]
Hence the optimal TWAP inventory is exactly
\begin{equation}
  \boxed{
    q_{\mathrm{TWAP}}^*(t)
    =
    Q\frac{T-t}{T}.
  }
  \label{eq:qstar-TWAP-bonus}
\end{equation}

\subsection{Final comparison with the mean--variance framework}

For both benchmarks, the expected-utility problem reduces to the deterministic
quadratic objective
\[
  J_B^{\mathrm{EU}}[q]
  =
  \eta\!\int_0^T \dot q_t^2 \dd t
  +
  \frac{\gamma\sigma^2}{2}\!\int_0^T (q_t-f_B(t))^2 \dd t,
\]
with
\[
  f_{\mathrm{TC}}(t)=Q,
  \qquad
  f_{\mathrm{TWAP}}(t)=Q\frac{T-t}{T}.
\]
Comparing with the mean--variance objective derived earlier in the report shows
that the two frameworks coincide after the identification
\[
  \boxed{\lambda = \frac{\gamma}{2}}.
\]

\begin{resultbox}[Final answer]
  In the expected-utility framework, the optimal strategies are
  \[
    q_{\mathrm{TC}}^*(t)
    =
    Q\left(1-\frac{\sinh(\kap t)}{\sinh(\kap T)}\right),
    \qquad
    q_{\mathrm{TWAP}}^*(t)
    =
    Q\frac{T-t}{T},
  \]
  with
  \[
    \kap = \sqrt{\frac{\gamma\sigma^2}{2\eta}}.
  \]
  The primitive HJB state is $(t,x,s,q)$ for TC and $(t,x,s,a,q)$ for TWAP.
\end{resultbox}
